local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local playerName = LocalPlayer.Name

-- Carrega a UI Library YoungStar
local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/bloodball/-back-ups-for-libs/main/youngstar"))()
local Window = library:CreateWindow("Remove ban")

-- Estado do Anti-Ban
local AntiBanAtivado = false

-- BotÃ£o Toggle do Anti-Ban
Window:Toggle("Anti-Ban", function(state)
    AntiBanAtivado = state
end)

-- FunÃ§Ã£o que remove banimentos e barreiras
local function desbanirJogador()
    local lots = workspace:FindFirstChild("001_Lots")
    if not lots then return end

    for _, house in pairs(lots:GetChildren()) do
        local picked = house:FindFirstChild("HousePickedByPlayer")
        if picked then
            local model = picked:FindFirstChild("HouseModel")
            if model then
                -- Remover de BannedPlayers
                local mailbox = model:FindFirstChild("001_MailBox")
                if mailbox then
                    local letter = mailbox:FindFirstChild("LetterFake")
                    if letter and letter:FindFirstChild("U1") then
                        local bannedPlayers = letter.U1:FindFirstChild("BannedPlayers")
                        if bannedPlayers then
                            for _, obj in pairs(bannedPlayers:GetChildren()) do
                                if obj.Name == playerName then
                                    obj:Destroy()
                                    print("âœ… Removido da lista de banidos de: " .. house.Name)
                                end
                            end
                        end
                    end
                end

                -- Destruir BannedBlock
                local bannedBlock = model:FindFirstChild("BannedBlock")
                if bannedBlock then
                    bannedBlock:Destroy()
                    print("ðŸšª BannedBlock destruÃ­do em: " .. house.Name)
                end
            end
        end
    end
end

-- Loop automÃ¡tico se ativado
task.spawn(function()
    while true do
        if AntiBanAtivado then
            pcall(desbanirJogador)
        end
        task.wait(3)
    end
end)
